## Slider Element

Sliders allow users to select a numeric value within a range. They support both integer and floating-point numbers.

<Tabs defaultValue="small">
    <TabsList>
        <TabsTrigger value="small">Basic</TabsTrigger>
        <TabsTrigger value="full">Full</TabsTrigger>
    </TabsList>
    <TabsContent value="small">
```luau
local Slider = Tab:Slider({
    Title = "Volume",
    Value = {
        Min = 0,
        Max = 100,
        Default = 50
    },
    Callback = function(value)
        print("Volume:", value)
    end
})
```
    </TabsContent>
    <TabsContent value="full">
```luau
local Slider = Tab:Slider({
    Title = "Effect Strength",
    Desc = "Slider description", -- optional
    Value = {
        Min = 0, -- minimum value
        Max = 100, -- maximum value
        Default = 50 -- initial value
    },
    Step = 1, -- increment step (use 0.1 for floats). optional
    Locked = false, -- disable slider. optional
    Flag = "my_slider", -- for config saving. optional
    Callback = function(value)
        print("Value changed:", value)
    end
})
```

## Slider Variants

### Integer Slider

```luau
Tab:Slider({
    Title = "Speed",
    Value = { Min = 1, Max = 10, Default = 5 },
    Step = 1, -- integer steps
    Callback = function(value)
        print("Speed:", value)
    end
})
```

### Float Slider

```luau
Tab:Slider({
    Title = "Sensitivity",
    Value = { Min = 0, Max = 1, Default = 0.5 },
    Step = 0.1, -- decimal steps
    Callback = function(value)
        print("Sensitivity:", value)
    end
})
```

### Large Range Slider

```luau
Tab:Slider({
    Title = "Population",
    Value = { Min = 0, Max = 1000000, Default = 500000 },
    Step = 10000,
    Callback = function(value)
        print("Population:", value)
    end
})
```

### Slider with Icons

```luau
Tab:Slider({
    Title = "Brightness",
    Icons = {
        From = "sfsymbols:sunMinFill",
        To = "sfsymbols:sunMaxFill",
    },
    Value = { Min = 0, Max = 100, Default = 75 },
    Callback = function(value)
        print("Brightness:", value)
    end
})
```
